# Handler class
class Handler:
  def new_message_handler(self, input_data, c2d):
    Logger.get_instance().set_trigger_name('new_message_handler')
    client_id = input_data['client']['id']
    dialog_id = input_data['message']['dialogID']
    Logger.get_instance().set_client_id(client_id).set_dialog_id(dialog_id)
    try:
      #check if it is answer for menu-question and fill client info(language)
      MenuManger.fillClienInfo(c2d,input_data, client_id, dialog_id)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: fillClienInfo fail: ' + str(e))
    Logger.get_instance().print_log()

  def client_updated_handler(self, input_data, c2d):
    Logger.get_instance().set_trigger_name('client_updated_handler')
    try:
      GroupManager.fillInfoForNewClient(c2d,input_data)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: fillInfoForNewClient fail: ' + str(e))
    Logger.get_instance().print_log()

  def before_sending_message_handler(self, input_data, c2d):
    Logger.get_instance().set_trigger_name('before_sending_message_handler')
    TagService.delete_tags_before_agent_message(c2d, input_data['client']['id'])
    Logger.get_instance().print_log_if_error()

  def after_closing_dialog_handler(self, input_data, c2d):
    Logger.get_instance().set_trigger_name('after_closing_dialog_handler')
    client_id = input_data['client']['id']
    dialog_id = input_data['dialog_id']
    client_info = c2d.get_client_info(client_id)
    dialog_info = Utils.get_dialog(c2d, dialog_id, None)
    request_id = dialog_info['last_request_id']
    Logger.get_instance().log_info('Set request_id from dialog info: ' + str(request_id))
    Logger.get_instance().set_client_id(client_id).set_dialog_id(dialog_id).set_request_id(request_id)
    try:
      AutoAssignService.reassign_after_closing_dialog(c2d,client_id,dialog_id)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: reassign_after_closing_dialog fail: ' + str(e))
    try:
      dialog = DialogManager.getDialogById(c2d, dialog_id)
      Logger.get_instance().set_channel_id(dialog['last_message']['channel_id'])
      relevant_dialog = DialogService.checkIfDialogIsRelevant(c2d,dialog)
      if relevant_dialog:
        DialogService.sendDialogToSalesforce(c2d,dialog,client_info)
      TagService.delete_tags_after_close(c2d, client_info, dialog_id)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: sendDialogToSalesforce fail: ' + str(e))
    Logger.get_instance().print_log()

  def auto_checking_handler(self, input_data, c2d):
    Logger.get_instance().set_trigger_name('auto_checking_handler')
    try:
      SLAService.autoclose_dialog_after_time(c2d, 600)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: autocheck_close_dialog_after_10min fail: ' + str(e))
    try:
      SLAService.set_tag_unanswered_dialogs(input_data, c2d, 180, 'FAIL_SLA')
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: mark_as_sla_fail_unanswer_dialogs fail: ' + str(e))
    try:
      SLAService.send_waiting_messasge(c2d)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: send_waiting_messasge fail: ' + str(e))
    try:
      SLAService.set_tag_unanswered_dialogs(input_data, c2d, 420, 'CS_NOTIF')
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: cs_notification_unanswer_dialogs fail: ' + str(e))
    try:
      AutoAssignService.auto_reassign_after_time(c2d, 30)
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: auto_reassign_after_time fail: ' + str(e))
    try:
      DialogService.send_dialogs_to_salesforce(c2d)
      print('try RESEND')
    except Exception as e:
      Logger.get_instance().log_error(c2d, 'ERROR: send_dialogs_to_salesforce fail: ' + str(e))
    Logger.get_instance().print_log_if_error()

  def dialog_transfer_handler(self, input_data, c2d):
    Logger.get_instance().set_trigger_name('dialog_transfer_handler')
    Logger.get_instance().set_client_id(input_data['client_id']).set_dialog_id(input_data['dialog_id'])
    Logger.get_instance().log_info('request_id from input data: ' + str(input_data['request_id']))
    try:

    is_transfered_to_group = Utils.is_transfered_to_group(c2d, dialog_id, client_id)

      transfered_queue = transfered_group_params[0]
      current_operator_group = None
      transfer_group = None
      Logger.get_instance().set_channel_id(channel_id)

          transfer_group = operator_group

      if current_operator_group != None and transfer_group != None:
        Logger.get_instance().log_info('current_operator_group id: ' + str(current_operator_group['id']))
        Logger.get_instance().log_info('current_operator_group name: ' + str(current_operator_group['name']))
        Logger.get_instance().log_info('transfer_group id: ' + str(transfer_group['id']))
        Logger.get_instance().log_info('transfer_group name: ' + str(transfered_queue))
        transfer_group_id = transfer_group['id']
        avialable_buisness_hours = Utils.checkBusinesshours(transfer_group_id)
        Logger.get_instance().log_info('avialable_buisness_hours : ' + str(avialable_buisness_hours))
        template_name = Constants.CHANNEL_NAME[str(channel_id)] + '_OUT'
        Logger.get_instance().log_info('template_name : ' + str(template_name))

          Utils.replace_tag(c2d, 'TS', 'CS', client_id, 'client')
          Utils.setClientCustomField(c2d, client_id, '5', 'CS')
          Logger.get_instance().log_info('Tag and field replaces from TS to CS')
          TagManager.delete_tags(c2d, client_info['id'], [Constants.TAG_NAME_ID['FAIL_SLA']])
          ClientManager.set_assigned_time_field(c2d, channel_id, client_info)
          free_operator_id = OperatorManager.find_free_online_operator_by_group(c2d, transfer_group_id)
          if free_operator_id != None:
            c2d.transfer_dialog(dialog_id, free_operator_id)

         Logger.get_instance().log_warn('Groups do not meet the conditions')
      else:
        Logger.get_instance().log_warn('Operator ' + str(last_operator_id) + 'dosnt have group')
    else:
      Logger.get_instance().log_info('The dialog is not assigned to a group')


# OperatorManager class
class OperatorManager:
  @staticmethod
  def find_free_online_operator_by_group(c2d, groupId):
    operators = c2d.get_online_operators()
    if len(operators) > 0:
      group_operators_ids = OperatorManager.get_operators_ids_by_group(c2d, groupId)
      default_operators_ids = OperatorManager.get_default_operators_ids()
      relevant_operators = []
      operators_dialogs = []
      free_operators = []
      for operator in operators:
        relevant_operator = operator['id'] in group_operators_ids and operator['id'] not in Constants.EXCEPTIONAL_AUTOASSIGN_OPERATORS and operator['id'] not in default_operators_ids and operator['offline_type'] == None and operator['opened_dialogs'] < Constants.MAX_OPENED_DIALOGS
        if relevant_operator:
          relevant_operators.append(operator)
          operators_dialogs.append(operator['opened_dialogs'])
      if len(relevant_operators) == 1:
        return relevant_operators[0]['id']
      elif len(operators_dialogs) > 0:
        minimum_dialogs = min(operators_dialogs)
        for operator in relevant_operators:
          if operator['opened_dialogs'] == minimum_dialogs:
            free_operators.append(operator['id'])
        if len(free_operators) > 0:
          random_operator = random.choice(free_operators)
          return random_operator
      else:
        return None
    return None

  @staticmethod
  def get_operators_ids_by_group(c2d, groupId):
    operator_groups = c2d.get_operators_groups()
    for operator_group in operator_groups:
      if operator_group['id'] == groupId:
        return operator_group['operator_ids']
    return []

  @staticmethod
  def check_operator_is_avialable(c2d, operator_id):
    online_operators = c2d.get_online_operators()
    for operator in online_operators:
      if operator['id'] == operator_id and operator['offline_type'] == None and operator['opened_dialogs'] < Constants.MAX_OPENED_DIALOGS:
        return True
    return False

  @staticmethod
  def get_online_operators_except_default(c2d):
    default_operators_ids = OperatorManager.get_default_operators_ids()
    online_operators = c2d.get_online_operators()
    avialable_operators = []
    for operator in online_operators:
      if operator['offline_type'] == None and operator['id'] not in default_operators_ids and operator['id'] not in Constants.EXCEPTIONAL_AUTOASSIGN_OPERATORS:
        avialable_operators.append(operator)
    avialable_operators_by_ids = Utils.map_by_field(avialable_operators, 'id')
    return avialable_operators_by_ids

  @staticmethod
  def get_default_operators_ids():
    default_operators_ids = []
    for group in Constants.DEFAULT_OPERATORS_BY_GROUP:
      default_operators_ids.append(Constants.DEFAULT_OPERATORS_BY_GROUP[group])
    return default_operators_ids

  @staticmethod
  def get_online_operators_mapped_by_group(c2d):
    online_operators = OperatorManager.get_online_operators_except_default(c2d)
    operators_groups = c2d.get_operators_groups()
    OperatorManager.get_online_operators_mapped_by_group(c2d, online_operators, operators_groups)


  @staticmethod
  def get_online_operators_mapped_by_group(c2d, online_operators, operators_groups):
    online_operators_by_group_id = {}
    for online_operator_id in online_operators:
      for operator_group in operators_groups:
        if online_operators[online_operator_id]['id'] in operator_group['operator_ids']:
          if operator_group['id'] not in online_operators_by_group_id:
            online_operators_by_group_id[operator_group['id']] = []
          online_operators_by_group_id[operator_group['id']].append(online_operators[online_operator_id])
    return online_operators_by_group_id

  def assignDialogOnGroup(c2d, client_id,channel_id, group_id, client_group_name, client_language,request_id, dialog_id, custom_fields):
    Logger.get_instance().add_method_name('GroupManager.assignDialogOnGroup')
    if dialog_id is None:
      messages = Utils.get_messages_by_request(c2d, request_id)
      if len(messages) > 0:
        dialog_id = messages[0]['dialogID']
    Logger.get_instance().set_request_id(request_id)
    Logger.get_instance().set_channel_id(channel_id)
    Logger.get_instance().set_client_id(client_id)
    Logger.get_instance().set_dialog_id(dialog_id)
    Logger.get_instance().log_info('client_group_name: ' + str(client_group_name))
    Logger.get_instance().log_info('client_language: ' + str(client_language))
    last_message_id = None
    is_exception_channel = int(channel_id) in Constants.EXCEPTION_CHANNELS

    if group_id is not None:
      business_hours = Utils.checkBusinesshours(group_id)
      # if not businessHours and group in substitution groups
      Logger.get_instance().log_info('General buisness hours: ' + str(business_hours))

      if client_group_name == 'TS' and client_language in Constants.TAG_NAME_ID:
        Utils.setTags(c2d,[Constants.TAG_NAME_ID[client_language], Constants.TAG_NAME_ID['TS']],'client',client_id)

      elif client_language in Constants.TAG_NAME_ID:
        result = Utils.setTags(c2d,[Constants.TAG_NAME_ID[client_language], Constants.TAG_NAME_ID['CS']],'client',client_id)
        Utils.setClientCustomField(c2d,client_id,'5','CS')  
      if business_hours or is_exception_channel:
        email_filled = custom_fields is not None and '1' in custom_fields and custom_fields['1'] is not None and custom_fields['1'] is not ''
        if client_group_name == 'TS'and email_filled :
          GroupManager.assign_to_ts(c2d, client_id, client_language, channel_id, dialog_id, group_id, request_id)
        elif email_filled:
          assign_tag = ClientService.check_request_has_tag(c2d, int(Constants.TAG_NAME_ID['ASSIGN']),request_id)
          AutoAssignService.assign_message_to_free_operator(c2d,assign_tag,request_id,client_id,int(group_id),client_language,dialog_id, channel_id, business_hours)
        elif is_exception_channel is True and email_filled is False:
          tag_language_email = Constants.tunnel_lgEmail[str(channel_id)]
          already_asked = ClientService.check_request_has_tag(c2d, Constants.TAG_NAME_ID[tag_language_email], request_id)
          if already_asked is False:
            Utils.setTags(c2d,[Constants.TAG_NAME_ID[tag_language_email]], 'request', request_id)

          Logger.get_instance().log_info('Email is empty. Start email question tunnel')
          Utils.setTags(c2d,[Constants.TAG_NAME_ID['email']], 'request', request_id)
          Utils.setTags(c2d,[Constants.TAG_NAME_ID['email']], 'client', client_id)
      else:
        template_Name = Constants.CHANNEL_NAME[str(channel_id)]+ '_OUT'
        Logger.get_instance().log_info('Buisness hours is: ' + str(business_hours))
        Logger.get_instance().log_info('Template name: ' + template_Name)
        TemplateManger.send_template(c2d,template_Name,client_language,client_id, channel_id)
        Utils.closeDialog(c2d,client_id, dialog_id)

      is_assigned_dialog_failed = True
      client_info = c2d.get_client_info(client_id)
      if '17' in client_info['custom_fields']:
        assignee_time_field = client_info['custom_fields']['17']
        assignee_time_by_channels = assignee_time_field.split(';')
        for assignee_time_by_channel in assignee_time_by_channels:
            assignee_time_string = assignee_time_by_channel.split(',')[1]
            assignee_time = datetime.strptime(assignee_time_string, Constants.DATE_FORMATS['DATE_F'])
            difference = datetime.now() - assignee_time
            total_seconds = difference.total_seconds()
            if total_seconds < time:
                is_assigned_dialog_failed = False
      return is_assigned_dialog_failed

    client_id = input_data['client_id']
    broker = None

    client = c2d.get_client_info(client_id)

    if '7' in client['custom_fields']:
      old_broker =  client['custom_fields']['7']

          broker = new_broker

          broker = old_broker + ',' + new_broker
      if new_broker in old_broker:
        broker = old_broker

    if (broker is None or not broker) and client_channel in Constants.CHANNEL_NAME:
      broker = Constants.CHANNEL_NAME[client_channel]
      Utils.setTags(c2d,[Constants.TAG_NAME_ID[broker]],'client',client_id)
    custom_fields = {'6':str(request_id), '7':broker} if broker is not None else {'6':str(request_id)}
    Utils.setClientCustomFields(c2d,client_id, custom_fields)

  @staticmethod
  def set_assigned_time_field(c2d, channel_id, client_info):
    Logger.get_instance().add_method_name('ClientManager.set_assigned_time_field')

    current_time_for_SLA = str(channel_id) + ',' + str(datetime.now())
    if '17' in client_info['custom_fields']:
        assignee_time_field = client_info['custom_fields']['17']
        assignee_time_by_channels = assignee_time_field.split(';')
        for assignee_time_by_channel in assignee_time_by_channels:
            if str(channel_id) == assignee_time_by_channel.split(',')[0]:
                assignee_time_by_channels.remove(assignee_time_by_channel) 
        assignee_time_by_channels.append(current_time_for_SLA)
        current_time_for_SLA = ';'.join(assignee_time_by_channels)         
    custom_fields = {'17':current_time_for_SLA}
    
    Utils.setClientCustomFields(c2d, client_info['id'], custom_fields)

    else:

  @staticmethod
  def prepareInfoWrapperForClosingDialog(c2d, dialog, client_info, request_id):
    Logger.get_instance().add_method_name('DialogService.prepareInfoWrapperForClosingDialog')
    transfered_from_ts = None
    experience = None
    purpose = None
    status = None
    strategy = None
    client_email = None
    client_phone = None 
    channel_name = None
    service = None
    service_category = None
    demanded_income = None
    language = None
    client_tags = []
    for client_tag in client_info['tags']: 
      client_tags.append(client_tag['id'])
    if int(Constants.TAG_NAME_ID['CONSULT']) in client_tags:
      status = 'Consulting'
    else:
      status = 'Short consulting'
    failed_sla_tag_id = Constants.TAG_NAME_ID['FAIL_SLA']
    channel_id = dialog['last_message']['channel_id']
    if request_id is None:
      request_id = dialog['last_message']['request_id']
    if str(channel_id) in Constants.SERVICE_FIELD_BY_BROKER and Constants.SERVICE_FIELD_BY_BROKER[str(channel_id)] in client_info['custom_fields']:
      field_number = Constants.SERVICE_FIELD_BY_BROKER[str(channel_id)]
      service_and_category = client_info['custom_fields'][field_number].split('//')
      if (len(service_and_category) > 0) :
        service = service_and_category[0]
      if len(service_and_category) > 1 :
        service_category = service_and_category[1]
    if '4' in client_info['custom_fields']:
      language = (client_info['custom_fields']['4']).lower()
    if '1' in client_info['custom_fields']:
      client_email = client_info['custom_fields']['1']
    if client_info['client_phone'] is not None and client_info['client_phone'] != '':
      client_phone = int(client_info['client_phone'])
    if str(channel_id) in Constants.CHANNEL_NAME:
      channel_name = Constants.CHANNEL_NAME[str(channel_id)]
    if '19' in client_info['custom_fields']:
      profiling_data = str(client_info['custom_fields']['19'])
      split_data = profiling_data.split(',')
      experience = split_data[0].split(': ')[1]
      purpose = split_data[1].split(': ')[1]
      status = split_data[2].split(': ')[1]
      strategy = split_data[3].split(': ')[1]
      demanded_income = split_data[4].split(': ')[1]
    if '18' in client_info['custom_fields']:
      sales_profiling_date = datetime.strptime(client_info['custom_fields']['18'], Constants.DATE_FORMATS['DATE_C2D']) - timedelta(hours=3)
      sales_profiling_date = datetime.strftime(sales_profiling_date, Constants.DATE_FORMATS['DATE_000Z'])
    else:
      sales_profiling_date = ''

    dialog_wrapper = {
      "transfered_from_ts": transfered_from_ts,
      "request_id" : str(request_id),
      "operator_id" : str(dialog['operator_id']),
      "failed_SLA" : True if int(failed_sla_tag_id) in client_tags else False,
      "dialog_id" : str(dialog['id']),
      "status" : status,
      "closed_date": str(datetime.now().isoformat()),
      "client":{
        "service_category": service_category,
        "service": service,
        "profiling": {
          "trading_purpose": purpose,
          "trading_frequency": status,
          "strategy":  strategy,
          "sales_profiling_date": str(sales_profiling_date),
          "experience": experience,
          "demanded_income": str(demanded_income)
        },
        "phone": str(client_phone),
        "name":client_info['assigned_name'],
        "id": str(client_info['id']),
        "email": str(client_email),
        "broker": channel_name if channel_name not in Constants.SF_BROKER_MAP else Constants.SF_BROKER_MAP[channel_name],
        "language": language
      },
      "channel_id" : str(channel_id)
    }
    return dialog_wrapper

class Utils:

  @staticmethod
  def to_str(arg):
    if not isinstance(arg, unicode):
      return str(arg)
    return unicode(arg).encode('utf-8')

  @staticmethod
  def getLanguageByLink(site_url):
    Logger.get_instance().add_method_name('Utils.getLanguageByLink')
    language = None
    for link_constant in Constants.LINK_LANGUAGE_CONSTANT:
      if link_constant in site_url:
        language = Constants.LINK_LANGUAGE_CONSTANT[link_constant]
        language = Constants.AVAILABLE_LANGUAGES[language]

    if(language == None):
      for url in Constants.AVAILABLE_URLS:
        if url in site_url:
          linkAfterDzone = site_url.split(url)[1]
          if len(linkAfterDzone) == 3 or (len(linkAfterDzone) == 4 and linkAfterDzone.endswith('/')):
            language = linkAfterDzone.split('/')[1]
          elif('/' in linkAfterDzone):
            language = linkAfterDzone.split('/')[1]
          if language in Constants.AVAILABLE_LANGUAGES and Constants.AVAILABLE_LANGUAGES[language.lower()] is not None:
            language = Constants.AVAILABLE_LANGUAGES[language.lower()]
          else:
            language = Constants.AVAILABLE_LANGUAGES['en']  
      
    Logger.get_instance().log_info('Language: ' + str(language))
    return language

  @staticmethod 
  def setTags(c2d, tags, tagType, clientId):

  @staticmethod 
  def checkBusinesshours(goup_id):
    Logger.get_instance().add_method_name('Utils.checkBusinesshours')
    if goup_id is not None:
      goup_id = int(goup_id)
      if goup_id in Constants.BUSINESS_HOURS:
        current_datetime = datetime.now()
        current_day_number = datetime.weekday(current_datetime)
        day_text = Constants.WEEK_DAYS[int(current_day_number)]
        if goup_id in Constants.BUSINESS_HOURS and day_text in Constants.BUSINESS_HOURS[goup_id]:
          business_hours_for_current_day = Constants.BUSINESS_HOURS[goup_id][day_text]
          if business_hours_for_current_day is not None:
            if current_datetime.time() > business_hours_for_current_day[0] and current_datetime.time() < business_hours_for_current_day[1]:
              Logger.get_instance().log_info('Business hours for group ' + str(goup_id) + ' is True')
              return True
      elif goup_id not in Constants.BUSINESS_HOURS:
          return True
    Logger.get_instance().log_info('Businesshours for group ' + str(goup_id) + 'is False')
    return False

  @staticmethod
  def get_param_from_parametric_url(c2d, client_id, param, client_messages):
    Logger.get_instance().add_method_name('Utils.get_param_from_parametric_url')
    if len(client_messages) > 0:
      first_client_message = client_messages[0]
      if first_client_message['transport'] in Constants.AVAILABLE_PARAMETRIC_MESSAGE_TRANSPORTS:
        message_text = first_client_message['text']
        encoded_text = None
        if first_client_message['transport'] == 'viber_public' or first_client_message['transport'] == 'telegram' and message_text is not None:
          splitted_text = message_text.split(' ')
          if len(splitted_text) > 1:
            encoded_text = splitted_text[1]
        elif first_client_message['transport'] == 'wa_dialog' and message_text is not None:
          splitted_text = message_text.split(':')
          if len(splitted_text) > 0:
            encoded_text = splitted_text[len(splitted_text)-1]
        elif first_client_message['transport'] == 'facebook'and message_text is not None:
          parsed_groups = re.findall('(?:ref: ?)(.*)', message_text)
          if len(parsed_groups) > 0:
            encoded_text = parsed_groups[0]

        if encoded_text is not None:
          params = None
          try:
            encoded_text = encoded_text + '=='
            base64_bytes = encoded_text.encode('ascii')
            message_bytes = base64.b64decode(base64_bytes)
            message = message_bytes.decode('ascii')
          
            params = json.loads(message)
          except Exception as e:
            Logger.get_instance().log_warn('Error convert to json parametric url')
          if params is not None and param in params:
            Logger.get_instance().log_info('Param ' + param + ' detected as: ' + str(params[param]))
            return params[param]
        Logger.get_instance().log_info('Param ' + param + ' is empty')
      else:
        Logger.get_instance().log_warn('Transport ' + str(first_client_message['transport']) + ' is not available in Constants')
    else:
      Logger.get_instance().log_warn('Client doesn\'t have any messages')
    return None

  @staticmethod
  def get_key_from_map_without_case(dictionary, text, ignore_case):
    if ignore_case is True:
      return next((value for key, value in dictionary.items() if key.lower().replace(' ', '')==text.lower().replace(' ', '')), None)
    else:
      return next((value for key, value in dictionary.items() if key==text), None)

class Logger:
  instance = None

  trigger_name = None
  client_id = None
  dialog_id = None
  request_id = None
  channel_id = None
  history = []

  has_errors = False

  @staticmethod
  def get_instance():
    if Logger.instance is None:
      Logger.instance = Logger()
    return Logger.instance

  def set_trigger_name(self, trigger_name):
    if self.trigger_name is None and trigger_name is not None:
      self.trigger_name = str(trigger_name)
    return self.instance

  def set_client_id(self, input_client_id):
    if self.client_id is None and input_client_id is not None:
      self.client_id = str(input_client_id)
    return self.instance

  def set_channel_id(self, input_channel_id):
    if self.channel_id is None and input_channel_id is not None:
      self.channel_id = str(input_channel_id)
    return self.instance

  def set_dialog_id(self, input_dialog_id):
    if self.dialog_id is None and input_dialog_id is not None:
      self.dialog_id = str(input_dialog_id)
    return self.instance

  def set_request_id(self, input_request_id):
    if self.request_id is None and input_request_id is not None:
      self.request_id = str(input_request_id)
    return self.instance

  def add_method_name(self, method_name):
    self.history.append('***' + method_name + '***')
    return self.instance

  def add_log(self, log_level, message):
    self.history.append('    ' + log_level + ': ' + message)

  def log_info(self, message):
    self.instance.add_log('INFO', message)

  def log_warn(self, message):
    self.instance.add_log('WARN', message)

  def log_error(self, c2d, message):
    admin_error = ''
    if self.trigger_name is not None:
      admin_error = admin_error + '*trigger_name*: ' + self.trigger_name + '\n'
    if self.client_id is not None:
      admin_error = admin_error + '*client_id*: ' + self.client_id + '\n'
    if self.dialog_id is not None:
      admin_error = admin_error + '*dialog_id*: ' + self.dialog_id + '\n'
    if self.request_id is not None:
      admin_error = admin_error + '*request_id*: ' + self.request_id + '\n'
    if self.channel_id is not None:
      admin_error = admin_error + '*channel_id*: ' + self.channel_id + '\n'
    Utils.logErrorToAdmin(c2d, admin_error + message)
    self.instance.add_log('ERROR', message)
    self.has_errors = True

  def print_log(self):
    if self.trigger_name is not None:
      print '*trigger_name*: ' + self.trigger_name
    if self.client_id is not None:
      print '*client_id*: ' + self.client_id
    if self.dialog_id is not None:
      print '*dialog_id*: ' + self.dialog_id
    if self.request_id is not None:
      print '*request_id*: ' + self.request_id
    if self.channel_id is not None:
      print '*channel_id*: ' + self.channel_id
    if self.history is not None:
      print json.dumps(self.history, indent = 4)

  def print_log_if_error(self):
    if self.has_errors is True:
      self.print_log()
